**方案二：数组的特殊处理**

      arr=[2,3,3,4,[2,3,4,[13,3,[3,4,6],4]]];
      arr=""+arr;            //这里对数组进行处理
      console.log(arr);      //2,3,3,4,2,3,4,13,3,3,4,6,4   //string
      arr=arr.split(",");   
      console.log(arr);      //["2", "3", "3", "4", "2", "3", "4", "13", "3", "3", "4", "6", "4"]  //array
      var arr1=[];
      for(var i=0;i<arr.length;i++){
         arr1[i]=eval(arr[i]);
      }
      console.log(arr1);     //[2, 3, 3, 4, 2, 3, 4, 13, 3, 3, 4, 6, 4]
